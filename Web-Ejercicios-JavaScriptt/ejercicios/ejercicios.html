Ejercicio 1 (condicional If)

Crea un programa que determine si un número es positivo o negativo.

                
                    function determinarPositivoNegativo(numero) {
                        if (numero > 0) {
                            return "El número es positivo";
                        } else if (numero < 0) {
                            return "El número es negativo";
                        } else {
                            return "El número es cero";
                        }
                    }
            
                    let numero = 5;
                    let resultado = determinarPositivoNegativo(numero);
                    console.log(resultado); // Imprimirá "El número es positivo"`;
